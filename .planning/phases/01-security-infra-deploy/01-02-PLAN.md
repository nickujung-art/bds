    import withSerwistInit from '@serwist/next'
    import { withSentryConfig } from '@sentry/nextjs'   // 추가

    const withSerwist = withSerwistInit({               // 기존 유지
      swSrc: 'src/app/sw.ts',
      swDest: 'public/sw.js',
      disable: process.env.NODE_ENV === 'development',
    })

    const nextConfig: NextConfig = {                   // 기존 headers() 그대로 유지
      async headers() {
        return [
          {
            source: '/(.*)',
            headers: [
              { key: 'Content-Security-Policy', value: [...].join('; ') },
              { key: 'X-Content-Type-Options', value: 'nosniff' },
              { key: 'X-Frame-Options', value: 'DENY' },
              { key: 'Referrer-Policy', value: 'strict-origin-when-cross-origin' },
            ],
          },
        ]
      },

      withSerwist(nextConfig),                          // withSerwist 안쪽 (Pitfall 3 방지)

    )

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| 앱 → Sentry.io | 에러/스택 트레이스 외부 전송 — 민감 데이터 포함 가능성 |
| 클라이언트 번들 → NEXT_PUBLIC_SENTRY_DSN | DSN이 브라우저에 노출됨 — 설계 상 의도된 공개값 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | Sentry 에러 페이로드 | mitigate | `enabled: process.env.NODE_ENV === 'production'`으로 개발 환경 노이즈 차단. 스택 트레이스에 PII 포함 주의 필요 (Phase 1 범위 내 처리) |
| T-02-02 | Information Disclosure | NEXT_PUBLIC_SENTRY_DSN 노출 | accept | Sentry DSN은 공개 설계값 — 수신 전용, 데이터 조회 불가. Sentry 프로젝트 레벨 rate limit으로 남용 방지 |
| T-02-03 | Repudiation | 에러 누락 (Sentry init 실패) | mitigate | instrumentation.ts의 NEXT_RUNTIME 조건부 로딩으로 서버/엣지 모두 커버. onRequestError = Sentry.captureRequestError로 요청 에러 캡처 |
| T-02-04 | Tampering | withSentryConfig 빌드 플러그인 | accept | 빌드 시 소스맵 업로드 (SENTRY_AUTH_TOKEN 없으면 스킵). 프로덕션 배포 전 토큰 설정 권장 |
</threat_model>

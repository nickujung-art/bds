/**
 * KAPT 단지 상세정보 적재 스크립트 (DATA-08)

import { createClient } from '@supabase/supabase-js'
import { fetchComplexList, fetchKaptBasicInfo } from '../src/services/kapt'

loadEnvConfig(process.cwd())

if (!process.env.KAPT_API_KEY) {
  console.error('KAPT_API_KEY 환경변수 필요')

if (!process.env.NEXT_PUBLIC_SUPABASE_URL || !process.env.SUPABASE_SERVICE_ROLE_KEY) {

  { auth: { autoRefreshToken: false, persistSession: false } },
)

**buildDongMap 함수:** fetchComplexList를 sgg_code별로 1회 호출하여 kaptCode → as3(dong) 매핑 Map 생성.

```typescript
async function buildDongMap(sggCodes: string[]): Promise<Map<string, string>> {
  const map = new Map<string, string>()
  for (const sggCode of sggCodes) {
    const list = await fetchComplexList(sggCode)
    for (const item of list) {
      if (item.as3) map.set(item.kaptCode, item.as3)
    }
    console.log(`  dong map: ${sggCode} → ${list.length}개 단지`)
  }
  return map
}
```

**main 함수 로직:**

1. regions 테이블에서 sgg_code, si, gu 조회 → regionMap (Map<string, {si, gu}>) 생성
2. buildDongMap(sggCodes) 호출 (6번 fetchComplexList API 호출)
3. complexes에서 WHERE kapt_code IS NOT NULL AND si IS NULL 조회 (id, kapt_code, sgg_code 컬럼)
4. 각 complex에 대해:
   a. fetchKaptBasicInfo(complex.kapt_code) 호출
   b. si/gu = regionMap.get(complex.sgg_code) ?? { si: null, gu: null }
   c. dong = dongMap.get(complex.kapt_code) ?? null
   d. built_year = kaptUsedate ? parseInt(kaptUsedate.slice(0, 4), 10) : null
   e. heat_type = info.heatType ?? info.codeHeatNm ?? null
   f. supabase.from('complexes').update({si, gu, dong, road_address: doroJuso, household_count: kaptdaCnt, built_year, heat_type, data_completeness: {transactions:false,school:false,kapt:true,poi:false}}).eq('id', complex.id)
   g. 100ms 대기 (KAPT rate limit 방어)
5. 진행 상황 출력: `[N/669] 단지명 ... done`
6. 완료 요약: 성공/실패 카운트
